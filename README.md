# 黑夜模式，比如：
	#import "WMNightManager.h"
	
    if ([WMNightManager currentStatus] == WMNightStatusNight) {
        [WMNightManager dawnComing];
    } else {
        [WMNightManager nightFalling];
    }
    
	titleLbl.textColor = kDayColor;
	titleLbl.nightTextColor = kNightColor;
